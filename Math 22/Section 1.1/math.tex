\begin{problem}{44} {Evaluate each of these expressions}{
  \problem{a}{$11000\land11011(01011\lor11011)$}{
    \begin{gather*}
      01011\lor11011 = 11011\\
      11000\land11011 = 11000\\
      11000\land11011(01011\lor11011) = 11000
    \end{gather*}
  }
  \problem{b}{$ (01111 \land 10101) \lor 01000$}{
    \begin{gather*}
      01111 \land 10101 = 00101\\
      00101 \lor 01000 = 01101\\
      (01111 \land 10101) \lor 01000 = 01101
    \end{gather*}
  }
  \problem{c}{$(01010 \oplus 11011) \oplus 01000$}{
    \begin{gather*}
      01010 \oplus 11011 = 10001\\
      10001 \oplus 01000 = 11001\\
      (01010 \oplus 11011) \oplus 01000 = 11001
    \end{gather*}
  }
  \problem{d}{$(11011 \lor 01010) \land (10001 \lor 11011) $}{

      11011 \lor 01010 = 11011\\

      (11011 \lor 01010) \land (10001 \lor 11011) = 11011

  }
