\begin{problem}{5} { Evaluate $\int_C(xz-y^2)ds$ where C is the line segment 
    from $(0, 1, 2)$ to $(-3, 7, -1)$ } {
        \step{x(t) &= -3t}{Reparameterize line segment}
        \step{y(t) &= 6t + 1}{}
        \step{z(t) &= -3t + 2}{}
        \step{\frac{dx}{dt} &= -3}{Derivatives with respect to t}
        \step{\frac{dy}{dt} &= 6}{}
        \step{\frac{dz}{dt} &= -3}{}
        \step{A &= \int_C f(x, y, z)ds}{Line integral with respect to arc length}
        \step{ds &= \sqrt{\left(\frac{dx}{dt}\right)^2 + \left(\frac{dy}{dt}\right)^2 + \left(\frac{dz}{dt}\right)^2}dt}
        {Arc length reparameterized in terms of t}
        \step{A &= \int_C f(x(t), y(t), z(t))\sqrt{\left(\frac{dx}{dt}\right)^2 + \left(\frac{dy}{dt}\right)^2 + \left(\frac{dz}{dt}\right)^2}dt}
            {Line integral with respect to t}
        \step{A &= \int_0^1 (-3t)(-3t + 2) - (6t + 1)^2\sqrt{(-3)^2 + (6)^2 + (-3)^2}dt}{Substitute}
        \step{A &= \int_0^1 \left((-3t)(-3t + 2) - (6t + 1)^2\right) (3\sqrt{6})dt}{Integrate}
        \step{A &= \int_0^1 (-27t^2 - 18t - 1)(3\sqrt{6})dt}{}
        \step{A &= 3\sqrt{6}\int_0^1 -27t^2 - 18t - 1dt}{}
        \step{A &= 3\sqrt{6} (-9t^3 - 9t^2 -t)|_0^1}{}
        \step{A &= -57\sqrt{6}}{}
    }{}
\end{problem}

\begin{problem}{6}{ Find the absolute maximum and minimum values of 
    $f(x, y) = x^4 + y^4 - 4xy + 2$ on the set 
    $D = \left\{(x,y) | 0 \leq x \leq 3, 0 \leq y \leq 3\right\}$.
    }{ 
        \step{D &= \left\{(x,y) | 0 \leq x \leq 3, 0 \leq y \leq 3\right\}}{Region is rectangular}
        \step{\frac{\partial f}{\partial x} &= 4x^3 - 4y}{1st partial derivative with respect to x}
        \step{\frac{\partial f}{\partial y} &= 4y^3 - 4x}{1st partial derivative with respect to y}
        \step{0 &= 4x^3 - 4y}{Set partials equal to 0 and evaluate system}
        \step{0 &= 4y^3 - 4x}{}
        \step{x &= y^3}{Solve for x}
        \step{0 &= 4(y^3)^3 - 4y}{Substitute}
        \step{0 &= 4y^9 - 4y}{Simplify}
        \step{0 &= 4y(y^8 - 1)}{Factor}
        \step{0 &= 4y(y^2 + 1)(y^2 - 1)}{Difference of 2 squares}
        \step{0 &= 4y(y^2 + 1)(y + 1)(y - 1)}{Difference of 2 squares}
        \step{y &= 0, 1, -1}{Real roots}
        \step{0 &= 4(0)^3 - 4x}{Let y = 0}
        \step{x &= 0}{}
        \step{0 &= 4(1)^3 - 4x}{Let y = 1}
        \step{x &= 1}{}
        \step{0 &= 4(-1)^3 - 4x}{Let y = -1}
        \step{x &= -1}{}
        \step{(x, y) &= (0, 0), (1, 1), (-1, -1)}{Critical Points}
        \step{f(0, 0) &= 2}{}
        \step{f(1, 1) &= 0}{}
        \step{f(-1, -1) &= 0}{$(-1, -1)$ out of bounds}
        \step{}{}
        \step{x &= 0}{Left edge of region}
        \step{f(0, y) &= y^4 + 2}{Let x = 0}
        \step{f'(0, y) &= 4y^3}{Differentiate with respect to y}
        \step{0 &= 4y^3}{Set to 0 and solve}
        \step{y &= 0}{}
        \step{f(0, 0) &= 2}{Bottom left corner, critical point (Local min)}
        \step{f(0, 2) &= 18}{Top left corner (Local max)}
        \step{}{}
        \step{x &= 3}{Right edge of region}
        \step{f(3, y) &= 81 + y^4 -12y + 2}{Let x = 3}
        \step{f'(3, y) &= 4y^3 - 12}{Differentiate with respect to y}
        \step{0 &= 4y^3 - 12}{Set to 0 and solve}
        \step{y &= \sqrt[3]{3}}{}
        }{}
    \end{problem}

\begin{problem}{8}{Use the method of Lagrange multipliers to find the maximum 
    and minimum values and their locations of $f(x, y) = xy$ subject to the 
    constraint $x^2 + y^2 - xy = 9$}{
        \step{\nabla f(x, y) &= \langle y, x \rangle}{Gradient of objective function}
        \step{\nabla g(x, y) &= \langle 2x-y, 2y-x \rangle}{Gradient of constraint function}
        \step{\nabla f(x, y) &= \lambda\nabla g(x, y)}{Lagrange multiplier formula}
        \step{\langle y, x \rangle &= \lambda\langle 2x-y, 2y-x \rangle}{}
        \step{E1: y &= \lambda(2x-y)}{Set up a system of equations}
        \step{E2: x &= \lambda(2y-x)}{}
        \step{E3: 9 &= x^2 + y^2 - xy}{}
        \step{E1: xy &= \lambda(2x^2-xy)}{Multiply both sides of E1 by x}
        \step{E2: xy &= \lambda(2y^2-xy)}{Multiply both sides of E2 by y}
        \step{\lambda(2x^2-xy) &= \lambda(2y^2-xy)}{$\therefore$ E1 = E2}
        \step{x^2 &= y^2}{}
        \step{x = y&,\ x = -y}{}
        \step{9 &= x^2 + x^2 - x^2}{Substitute x = y into E3}
        \step{\pm3 &= x}{}
        \step{9 &= 9 + y^2 - 3y}{Substitute x = 3 into E3}
        \step{0 &= y(y - 3)}{}
        \step{y = 0&,\ y = 3}{}
        \step{(3, -3)&, (-3, 3)}{Local Extrema}
        \step{f(3, -3) &= f(-3, 3) = -9}{}
        \step{Let\ x &= 1}{Find a point within the constraint to determine if extrema are}
        \step{9 &= 1 + y^2 - y}{minimum or maximum}
        \step{y \approx -2.37&,\ y \approx 3.37}{}
        \step{f(2.37, -2.37) &> -9}{$\therefore f(3, -3)$ and $f(-3, 3)$ are local minimums}
    }{}
\end{problem}
